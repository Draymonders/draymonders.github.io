      ## жүҖи°“дјҳеҢ–пјҢдҫӢеҰӮзңҒз•ҘжҺүд»Јз Ғдёӯд»ҺжңӘдҪҝз”ЁиҝҮзҡ„еҸҳйҮҸгҖҒзӣҙжҺҘе°ҶеёёйҮҸиЎЁиҫҫејҸз”Ёз»“жһңеҖјд»ЈжӣҝзӯүзӯүпјҢиҝҷдәӣж“ҚдҪңдјҡзј©еҮҸзӣ®ж Үж–Үд»¶жүҖеҢ…еҗ«зҡ„д»Јз ҒйҮҸпјҢжҸҗй«ҳжңҖз»Ҳз”ҹжҲҗзҡ„еҸҜжү§иЎҢж–Үд»¶зҡ„иҝҗиЎҢж•ҲзҺҮгҖӮ

# -O йҖүйЎ№е‘ҠиҜү g++ еҜ№жәҗд»Јз ҒиҝӣиЎҢеҹәжң¬дјҳеҢ–гҖӮиҝҷдәӣдјҳеҢ–еңЁеӨ§еӨҡж•°жғ…еҶөдёӢйғҪдјҡдҪҝзЁӢеәҸжү§иЎҢзҡ„жӣҙеҝ«гҖӮ -O2 йҖүйЎ№е‘ҠиҜү g++ дә§з”ҹе°ҪеҸҜиғҪе°Ҹе’Ңе°ҪеҸҜиғҪеҝ«зҡ„д»Јз ҒгҖӮ еҰӮ-O2пјҢ-O3пјҢ-OnпјҲn еёёдёә0вҖ“3пјү
# -O еҗҢж—¶еҮҸе°Ҹд»Јз Ғзҡ„й•ҝеәҰе’Ңжү§иЎҢж—¶й—ҙпјҢе…¶ж•Ҳжһңзӯүд»·дәҺ-O1
# -O0 иЎЁзӨәдёҚеҒҡдјҳеҢ–
# -O1 дёәй»ҳи®ӨдјҳеҢ–
# -O2 йҷӨдәҶе®ҢжҲҗ-O1зҡ„дјҳеҢ–д№ӢеӨ–пјҢиҝҳиҝӣиЎҢдёҖдәӣйўқеӨ–зҡ„и°ғж•ҙе·ҘдҪңпјҢеҰӮжҢҮд»Өи°ғж•ҙзӯүгҖӮ
# -O3 еҲҷеҢ…жӢ¬еҫӘзҺҜеұ•ејҖе’Ңе…¶д»–дёҖдәӣдёҺеӨ„зҗҶзү№жҖ§зӣёе…ізҡ„дјҳеҢ–е·ҘдҪңгҖӮ
# йҖүйЎ№е°ҶдҪҝзј–иҜ‘зҡ„йҖҹеәҰжҜ”дҪҝз”Ё -O ж—¶ж…ўпјҢ дҪҶйҖҡеёёдә§з”ҹзҡ„д»Јз Ғжү§иЎҢйҖҹеәҰдјҡжӣҙеҝ«гҖӮ

    ## д»ҘдёӢе‘Ҫд»ӨеҗҺжӢ¬еҸ·еҶ…дёәе‘Ҫд»Өзҡ„з®ҖеҢ–дҪҝз”ЁпјҢжҜ”еҰӮrunпјҲrпјүпјҢзӣҙжҺҘиҫ“е…Ҙе‘Ҫд»Ө r е°ұд»ЈиЎЁе‘Ҫд»Өrun

$(gdb)help(h)        # жҹҘзңӢе‘Ҫд»Өеё®еҠ©пјҢе…·дҪ“е‘Ҫд»ӨжҹҘиҜўеңЁgdbдёӯиҫ“е…Ҙhelp + е‘Ҫд»Ө 

$(gdb)run(r)         # йҮҚж–°ејҖе§ӢиҝҗиЎҢж–Үд»¶пјҲrun-textпјҡеҠ иҪҪж–Үжң¬ж–Үд»¶пјҢrun-binпјҡеҠ иҪҪдәҢиҝӣеҲ¶ж–Үд»¶пјү

$(gdb)start          # еҚ•жӯҘжү§иЎҢпјҢиҝҗиЎҢзЁӢеәҸпјҢеҒңеңЁз¬¬дёҖиЎҢжү§иЎҢиҜӯеҸҘ

$(gdb)list(l)        # жҹҘзңӢеҺҹд»Јз ҒпјҲlist-n,д»Һз¬¬nиЎҢејҖе§ӢжҹҘзңӢд»Јз ҒгҖӮlist+ еҮҪж•°еҗҚпјҡжҹҘзңӢе…·дҪ“еҮҪж•°пјү

$(gdb)set            # и®ҫзҪ®еҸҳйҮҸзҡ„еҖј

$(gdb)next(n)        # еҚ•жӯҘи°ғиҜ•пјҲйҖҗиҝҮзЁӢпјҢеҮҪж•°зӣҙжҺҘжү§иЎҢпјү

$(gdb)step(s)        # еҚ•жӯҘи°ғиҜ•пјҲйҖҗиҜӯеҸҘпјҡи·іе…ҘиҮӘе®ҡд№үеҮҪж•°еҶ…йғЁжү§иЎҢпјү

$(gdb)backtrace(bt)  # жҹҘзңӢеҮҪж•°зҡ„и°ғз”Ёзҡ„ж Ҳеё§е’ҢеұӮзә§е…ізі»

$(gdb)frame(f)       # еҲҮжҚўеҮҪж•°зҡ„ж Ҳеё§

$(gdb)info(i)        # жҹҘзңӢеҮҪж•°еҶ…йғЁеұҖйғЁеҸҳйҮҸзҡ„ж•°еҖј

$(gdb)finish         # з»“жқҹеҪ“еүҚеҮҪж•°пјҢиҝ”еӣһеҲ°еҮҪж•°и°ғз”ЁзӮ№

$(gdb)continue(c)    # з»§з»ӯиҝҗиЎҢ

$(gdb)print(p)       # жү“еҚ°еҖјеҸҠең°еқҖ

$(gdb)quit(q)        # йҖҖеҮәgdb

$(gdb)break+num(b)                 # еңЁз¬¬numиЎҢи®ҫзҪ®ж–ӯзӮ№

$(gdb)info breakpoints             # жҹҘзңӢеҪ“еүҚи®ҫзҪ®зҡ„жүҖжңүж–ӯзӮ№

$(gdb)delete breakpoints num(d)    # еҲ йҷӨз¬¬numдёӘж–ӯзӮ№

$(gdb)display                      # иҝҪиёӘжҹҘзңӢе…·дҪ“еҸҳйҮҸеҖј

$(gdb)undisplay                    # еҸ–ж¶ҲиҝҪиёӘи§ӮеҜҹеҸҳйҮҸ

$(gdb)watch                        # иў«и®ҫзҪ®и§ӮеҜҹзӮ№зҡ„еҸҳйҮҸеҸ‘з”ҹдҝ®ж”№ж—¶пјҢжү“еҚ°жҳҫзӨә

$(gdb)i watch                      # жҳҫзӨәи§ӮеҜҹзӮ№

$(gdb)enable breakpoints           # еҗҜз”Ёж–ӯзӮ№

$(gdb)disable breakpoints          # зҰҒз”Ёж–ӯзӮ№

$(gdb)x                            # жҹҘзңӢеҶ…еӯҳx/20xw жҳҫзӨә20дёӘеҚ•е…ғпјҢ16иҝӣеҲ¶пјҢ4еӯ—иҠӮжҜҸеҚ•е…ғ

$(gdb)run argv[1] argv[2]          # и°ғиҜ•ж—¶е‘Ҫд»ӨиЎҢдј еҸӮ

$(gdb)set follow-fork-mode child   # MakefileйЎ№зӣ®з®ЎзҗҶпјҡйҖүжӢ©и·ҹиёӘзҲ¶еӯҗиҝӣзЁӢпјҲfork()пјү
