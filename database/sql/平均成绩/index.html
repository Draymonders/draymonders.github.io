  • 平均成绩

成绩表SC，其中语文编号为1（S，C，score）S学生编号、C课程编号、score分数 找出 语文成绩及格、平均成绩不及格的学生编号

create table SC (
    S int,
    C int,
    score int
);

insert into SC(`S`, `C`, `score`) values 
    (1, 1, 90),
    (1, 2, 80),
    (1, 3, 90),
    (2, 1, 60),
    (2, 2, 30),
    (2, 3, 30),
    (3, 1, 80),
    (3, 2, 0),
    (3, 3, 0),
    (4, 1, 10),
    (4, 2, 10),
    (4, 3, 10);

select S from SC 
where C = 1 and score >= 60 and S in (
        select S from SC 
        group by S 
        having avg(score) < 60
    );
