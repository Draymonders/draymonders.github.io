class Solution {
public:

    void dfs(string& str, vector<string>& res, int st, int ed) {
        if (st >= ed) {
            res.push_back(str);
            return ;
        }
        unordered_map<char, int> mp;
        for (int i=st; i<=ed; i++) {
            if (mp.count(str[i])) {
                continue;
            }

            swap(str[i], str[st]);
            dfs(str, res, st+1, ed);
            mp[str[i]] ++;

            swap(str[i], str[st]);
        }
    }

    vector<string> Permutation(string str) {
        sort(str.begin(), str.end());
        int n = str.size();
        vector<string> res;
        dfs(str, res, 0, n-1);
        sort(res.begin(), res.end(), [](string& a, string& b) {
            return a < b;
        });
        return res;
    }
};
