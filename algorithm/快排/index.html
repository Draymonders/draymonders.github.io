class Solution {
public:

    void quickSort(vector<int>& arr, int l, int r) {
        if (l >= r) return ;
        int i = l, j = r;
        int v = arr[l];
        while (i < j) {
            while (i < j && arr[j] >= v) j--;
            while (i < j && arr[i] <= v) i++;
            if (i < j) {
                swap(arr[i], arr[j]);
            }
        }
        swap(arr[l], arr[i]);
        quickSort(arr, l, i-1);
        quickSort(arr, i+1, r);
    }

    /**
     * @param arr int整型vector 待排序的数组
     * @return int整型vector
     */
    vector<int> MySort(vector<int>& arr) {
        int n = arr.size();
        quickSort(arr, 0, n-1);
        return arr;
    }
};
