    #include <bits/stdc++.h>
using namespace std;

void heapDown(vector<int>& nums, int pos, int lim) {
    while (pos <= lim) {
        int next = pos;
        if (pos*2+1 <= lim) {
            int& s1 = nums[pos*2+1];
            if (s1 < nums[pos]) {
                swap(s1, nums[pos]);
                next = pos*2+1;
            }
        }
        if (pos*2+2 <= lim) {
            int& s2 = nums[pos*2+2];
            if (s2 < nums[pos]) {
                swap(s2, nums[pos]);
                next = pos*2+2;
            }
        }
        if (next == pos) 
            break;
        pos = next;
    }
    return ;
}

void heapUp(vector<int>& nums, int pos) {
    while (pos > 0) {
        int next = pos;
        if (pos / 2 >= 0) {
            int& fa = nums[pos/2];
            if (nums[pos] < fa) {
                swap(nums[pos], fa);
                next = pos / 2;
            }
        }
        if (next == pos) 
            break;
        pos = next;
    }
}

void heapSort(vector<int>& nums) {
    int n = nums.size();
    if (n <= 1) {
        return ;
    }
    // е…ҲжҠҠеҶ…йғЁз»“жһ„еҸҳжҲҗе Ҷ
    for (int i=n/2-1; i>=0; i--) {
        // жҜҸдёӘйқһеҸ¶еӯҗиҠӮзӮ№иҝӣиЎҢеӯҗж ‘еҶ…зҡ„и°ғж•ҙ
        heapDown(nums, i, n-1);
    }

    while (n > 1) {
        swap(nums[0], nums[n-1]);
        n--;
        heapDown(nums, 0, n-1);
    }
}

// ж·»еҠ е…ғзҙ еҲ°е Ҷдёӯ
void addToHeap(vector<int>& nums, int val) {
    int n = nums.size();
    nums.push_back(val);
    heapUp(nums, n);
}

void print(vector<int>& v) {
    for (auto x : v) {
        cout << x << " ";
    }
    cout << endl;
}

int main() 
{
    vector<int> v = {7,8,3,2,1,7};
    heapSort(v);

    addToHeap(v, 4);
    print(v);
    return 0;
}
