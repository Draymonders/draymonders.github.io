归并排序

逆序对

在数组中的两个数字，如果前面一个数字大于后面的数字，则这两个数字组成一个逆序对。 输入一个数组,求出这个数组中的逆序对的总数P。并将P对1000000007取模的结果输出。 即输出P%1000000007

int mod = 1000000007;

class Solution {
public:
    int res = 0;
    vector<int> temp;

    void mergeSort(vector<int>& data, int l, int r) {
        if (l >= r) return ;
        // printf("%d %d\n", l, r);
        int m = (l + r) /2;
        mergeSort(data, l, m);
        mergeSort(data, m+1, r);
        // merge
        for (int i=l; i<=r; i++) 
            temp[i] = data[i];
        int i = l, j = m+1;
        int k = l;
        while (i <= m && j <= r) {
            if (temp[i] <= temp[j]) {
                data[k ++] = temp[i];
                i++;
            } else {
                data[k ++] = temp[j];
                res = (res + m - i + 1) % mod;
                j ++;
            }
        }
        while (i <= m) {
            data[k++] = temp[i];
            i++;
        }
        while (j <= r) {
            data[k++] = temp[j];
            j++;
        }
    }

    int InversePairs(vector<int> data) {
        res = 0;
        temp.resize(data.size() + 1);

        mergeSort(data, 0, data.size()-1);
        return res;
    }
};
