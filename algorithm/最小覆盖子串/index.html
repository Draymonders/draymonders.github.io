class Solution {
public:

    string minWindow(string s, string t) {
        // mp1记录s的区间字符信息，mp2存t整串的字符信息
        unordered_map<char, int> mp1, mp2;
        // size记录有多少种字符
        int size = 0;
        for (auto ch : t) {
            if (mp2.count(ch) == 0) {
                size ++;
            }
            mp2[ch] ++;
        }

        int l = 0, r = 0;
        int n = s.size();

        // 起点和最小长度
        int st = -1;
        int mn = n + 1;
        // sz记录满足区间情况的字符数
        int sz = 0;

        while (r < n) {
            mp1[s[r]] ++;
            if (mp1[s[r]] == mp2[s[r]]) {
                sz ++;
            }
            while (sz == size && l <= r && mp1[s[l]] >= mp2[s[l]]) {
                if (r - l + 1 < mn) {
                    st = l;
                    mn = r - l + 1;
                }
                mp1[s[l]]--;
                if (mp1[s[l]] < mp2[s[l]]) {
                    sz --;
                }
                l ++;
            }
            r++;
        }
        return mn > n ? "" : s.substr(st, mn);
    }
};
