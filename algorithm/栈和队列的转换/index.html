两个栈实现队列

反转两次

class Solution
{
public:
    void push(int x) {
        s1.push(x);
    }

    int pop() {
        if (s2.empty()) {
           while (!s1.empty()) {
            int x = s1.top(); s2.push(x); s1.pop();
           }
        }
        int x = s2.top(); s2.pop();
        return x;
    }

private:
    stack<int> s1;
    stack<int> s2;
};

两个队列实现栈

两个队列 - 新增元素时，放到q1队首, 然后将q2的所有元素依次放入q1，最后所有元素再放到q2 - 获取栈顶时，直接从q2首部去拿

class MyStack {
public:
    queue<int> q1, q2;
    /** Initialize your data structure here. */
    MyStack() {
        while (!q1.empty()) 
            q1.pop();
        while (!q2.empty()) 
            q2.pop();
    }

    /** Push element x onto stack. */
    void push(int x) {
        q1.push(x);
        while (!q2.empty()) {
            int x = q2.front(); q2.pop();
            q1.push(x);
        }
        while (!q1.empty()) {
            int x = q1.front(); q1.pop();
            q2.push(x);
        }
    }

    /** Removes the element on top of the stack and returns that element. */
    int pop() {
        if (q2.empty()) {
            return -1;
        }
        int x = q2.front(); q2.pop();
        return x;
    }

    /** Get the top element. */
    int top() {
        if (q2.empty()) {
            return -1;
        }
        return q2.front();
    }

    /** Returns whether the stack is empty. */
    bool empty() {
        return q2.empty();
    }
};
