是否存在环

class Solution {
public:
    bool hasCycle(ListNode *head) {
        ListNode* fast = head;
        ListNode* slow = head;
        while (fast != NULL) {
            slow = slow->next;
            if (fast -> next) {
                fast = fast->next->next;    
            } else {
                return false;
            }
            if (fast == slow)
                return true;
        }
        return fast != NULL ? true : false; 
    }
};
