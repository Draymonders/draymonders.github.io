旋转数组

左旋

class Solution {
public:

    void reverse(string& s, int l, int r) {
        while (l <= r) {
            swap(s[l], s[r]);
            l++, r--;
        }
    }

    string reverseLeftWords(string s, int k) {
        int n = s.size();
        k %= n;
        reverse(s, 0, k-1);
        reverse(s, k, n-1);
        reverse(s, 0, n-1);
        return s;
    }
};

旋转数组找某一个值

最小值

O(logn)求得旋转数组的最小值，包含重复

class Solution {
public:
    int findMin(vector<int>& nums) {
        int n = nums.size();

        int l = 0, r = n-1;
        int tmp = nums[0];

        int ans = -1;
        while (r >= 0 && nums[r] == tmp) r--;

        while (l <= r) {
            int m = (l + r) / 2;
            int x = nums[m];
            if (x < tmp) 
                ans = x, r = m-1;
            else 
                l = m + 1;
        }
        if (ans == -1)
            ans = nums[0];
        return ans;
    }
};

找某个值

class Solution {
public:
    int search(vector<int>& nums, int target) {
        if (nums.size() == 0)
            return -1;

        int n = nums.size();

        int l = 0, r = n-1;
        int ans = -1;
        int tmp = nums[0];
        while (r >= 0 && nums[r] == tmp) r--;

        while (l <= r) {
            int m = (l + r) / 2;
            // nums[m]在右边
            if (nums[m] < tmp) {
                // target在左边
                if (target >= tmp) {
                    r = m-1;
                } else {
                    // 在同一边
                    if (nums[m] >= target) {
                        ans =m, r = m-1;
                    } else {
                        l = m+1;
                    }
                }
            } else {
                // nums[m]在左边
                // target在右边
                if (target < tmp) {
                    l = m + 1;
                } else {
                    // 在同一边
                    if (nums[m] >= target) {
                        ans =m, r = m-1;
                    } else {
                        l = m+1;
                    }
                }
            }
        }
        if (ans != -1 && nums[ans] == target)
            return ans;
        return -1;
    }
};
