/**
 * struct TreeNode {
 *  int val;
 *  struct TreeNode *left;
 *  struct TreeNode *right;
 * };
 */

class Solution {
public:

    void preOrder(TreeNode* root, vector<int>& pre) {
        if (root == NULL) {
            return ;
        }
        TreeNode* cur = root;
        stack<TreeNode*> st;
        while (cur || !st.empty()) {
            if (cur) {
                pre.push_back(cur->val);
                st.push(cur);
                cur = cur->left; 
            } else {
                cur = st.top();
                st.pop();
                cur = cur->right;
            }
        }
        return ;
    }

    void inOrder(TreeNode* root, vector<int>& in) {
        if (root == NULL) {
            return ;
        }
        TreeNode* cur = root;
        stack<TreeNode*> st;
        while (cur || !st.empty()) {
            if (cur) {
                st.push(cur);
                cur = cur->left; 
            } else {
                cur = st.top();
                in.push_back(cur->val);
                st.pop();
                cur = cur->right;
            }
        }
        return ;
    }

    // 后序遍历需要注意
    void backOrder(TreeNode* root, vector<int>& back) {
        if (root == NULL) {
            return ;
        }
        TreeNode* cur = root;
        stack<pair<TreeNode*, int>> st;
        while (cur || !st.empty()) {
            if (cur) {
                st.push({cur, 1});
                cur = cur->left; 
            } else {
                pair<TreeNode*, int> temp = st.top(); st.pop();
                if (temp.second == 2) {
                    back.push_back(temp.first->val);
                    cur = NULL;
                } else {
                    st.push({temp.first, 2});
                    cur = temp.first->right;
                }
            }
        }
        return ;
    }

    /**
     * 
     * @param root TreeNode类 the root of binary tree
     * @return int整型vector<vector<>>
     */
    vector<vector<int> > threeOrders(TreeNode* root) {
        // write code here
        vector<vector<int>> res;
        vector<int> pre;
        vector<int> in;
        vector<int> back;
        preOrder(root, pre);
        inOrder(root, in);
        backOrder(root, back);
        res.push_back(pre);
        res.push_back(in);
        res.push_back(back);
        return res;
    }
};

层序遍历

/**
 * struct TreeNode {
 *  int val;
 *  struct TreeNode *left;
 *  struct TreeNode *right;
 * };
 */

class Solution {
public:
    /**
     * 
     * @param root TreeNode类 
     * @return int整型vector<vector<>>
     */
    vector<vector<int> > levelOrder(TreeNode* root) {
        if (root == NULL) {
            return {};
        }
        vector<vector<int>> res;
        queue<TreeNode*> que;
        que.push(root);
        while (!que.empty()) {
            vector<int> temp;
            int sz = que.size();
            for (int i=0; i<sz; i++) {
                TreeNode* f = que.front();
                que.pop();
                temp.push_back(f->val);
                if (f->left) {
                    que.push(f->left);
                } 
                if (f->right) {
                    que.push(f->right);
                }
            }
            res.push_back(temp);
        }
        return res;
    }
};
